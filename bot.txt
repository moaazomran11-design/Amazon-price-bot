import os
import re
import sqlite3
import asyncio
import logging
import httpx
from bs4 import BeautifulSoup
from telegram import Update, InlineKeyboardButton, InlineKeyboardMarkup
from telegram.ext import (
    Application, CommandHandler, MessageHandler,
    CallbackQueryHandler, ContextTypes, filters
)

# ==================== الإعدادات ====================
TOKEN = "ضع_التوكن_هنا"  # ← ضع التوكن الجديد هنا
CHECK_INTERVAL = 3600  # فحص كل ساعة (بالثواني)

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

HEADERS = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36",
    "Accept-Language": "ar-EG,ar;q=0.9,en;q=0.8",
    "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
}

# ==================== قاعدة البيانات ====================
def init_db():
    conn = sqlite3.connect("prices.db")
    c = conn.cursor()
    c.execute("""
        CREATE TABLE IF NOT EXISTS watchlist (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            user_id INTEGER,
            url TEXT,
            product_name TEXT,
            current_price REAL,
            target_price REAL,
            last_price REAL
        )
    """)
    conn.commit()
    conn.close()

def get_conn():
    return sqlite3.connect("prices.db")

# ==================== سكرابينج الأسعار ====================
async def get_price(url: str):
    try:
        async with httpx.AsyncClient(headers=HEADERS, follow_redirects=True, timeout=15) as client:
            r = await client.get(url)
            soup = BeautifulSoup(r.text, "html.parser")

            # اسم المنتج
            name_tag = soup.find("span", {"id": "productTitle"})
            name = name_tag.get_text(strip=True) if name_tag else "منتج غير معروف"

            # السعر - عدة محاولات
            price = None
            selectors = [
                ("span", {"class": "a-price-whole"}),
                ("span", {"id": "priceblock_ourprice"}),
                ("span", {"id": "priceblock_dealprice"}),
                ("span", {"class": "a-offscreen"}),
            ]
            for tag, attrs in selectors:
                el = soup.find(tag, attrs)
                if el:
                    raw = el.get_text(strip=True).replace(",", "").replace("ج.م", "").replace("EGP", "").strip()
                    nums = re.findall(r"[\d.]+", raw)
                    if nums:
                        price = float(nums[0])
                        break

            return name, price
    except Exception as e:
        logger.error(f"Error scraping {url}: {e}")
        return None, None

def extract_asin(url: str):
    match = re.search(r"/dp/([A-Z0-9]{10})", url)
    return match.group(1) if match else None

def clean_url(url: str):
    asin = extract_asin(url)
    if asin:
        return f"https://www.amazon.eg/dp/{asin}"
    return url

# ==================== الأوامر ====================
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "👋 أهلاً! أنا بوت مراقبة أسعار Amazon.eg\n\n"
        "📌 الأوامر المتاحة:\n"
        "/add - إضافة منتج للمراقبة\n"
        "/list - عرض منتجاتك\n"
        "/help - المساعدة"
    )

async def help_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "🛒 طريقة الاستخدام:\n\n"
        "1️⃣ انسخ لينك المنتج من Amazon.eg\n"
        "2️⃣ ابعت /add\n"
        "3️⃣ الصق اللينك\n"
        "4️⃣ حدد السعر المستهدف\n\n"
        "✅ البوت هيبعتلك إشعار لما السعر ينزل للمستهدف!"
    )

async def add_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "🔗 ابعت لينك المنتج من Amazon.eg:"
    )
    context.user_data["waiting_for"] = "url"

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.strip()
    waiting = context.user_data.get("waiting_for")

    if waiting == "url":
        if "amazon.eg" not in text and "amazon.com" not in text:
            await update.message.reply_text("❌ اللينك مش صح، ابعت لينك من Amazon.eg")
            return
        url = clean_url(text)
        context.user_data["pending_url"] = url
        context.user_data["waiting_for"] = "target_price"
        msg = await update.message.reply_text("⏳ بجيب معلومات المنتج...")
        name, price = await get_price(url)
        if price is None:
            await msg.edit_text("❌ مقدرتش أجيب السعر، تأكد من اللينك وحاول تاني")
            context.user_data.clear()
            return
        context.user_data["pending_name"] = name
        context.user_data["pending_price"] = price
        await msg.edit_text(
            f"✅ المنتج: {name[:60]}...\n"
            f"💰 السعر الحالي: {price:.2f} ج.م\n\n"
            f"🎯 ابعت السعر المستهدف (رقم فقط):"
        )

    elif waiting == "target_price":
        try:
            target = float(text.replace(",", ""))
        except ValueError:
            await update.message.reply_text("❌ ابعت رقم صح مثلاً: 150")
            return

        url = context.user_data["pending_url"]
        name = context.user_data["pending_name"]
        price = context.user_data["pending_price"]
        user_id = update.effective_user.id

        conn = get_conn()
        conn.execute(
            "INSERT INTO watchlist (user_id, url, product_name, current_price, target_price, last_price) VALUES (?,?,?,?,?,?)",
            (user_id, url, name, price, target, price)
        )
        conn.commit()
        conn.close()
        context.user_data.clear()

        await update.message.reply_text(
            f"✅ تمت الإضافة!\n\n"
            f"📦 {name[:60]}...\n"
            f"💰 السعر الحالي: {price:.2f} ج.م\n"
            f"🎯 السعر المستهدف: {target:.2f} ج.م\n\n"
            f"هبعتلك إشعار لما السعر ينزل! 🔔"
        )
    else:
        if text.startswith("http"):
            context.user_data["waiting_for"] = "url"
            await handle_message(update, context)

async def list_cmd(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    conn = get_conn()
    rows = conn.execute(
        "SELECT id, product_name, current_price, target_price, url FROM watchlist WHERE user_id=?",
        (user_id,)
    ).fetchall()
    conn.close()

    if not rows:
        await update.message.reply_text("📭 مفيش منتجات في قائمتك\nابعت /add لإضافة منتج")
        return

    for row in rows:
        pid, name, cur, target, url = row
        emoji = "✅" if cur <= target else "⏳"
        keyboard = [[
            InlineKeyboardButton("🗑 حذف", callback_data=f"del_{pid}"),
            InlineKeyboardButton("🔗 فتح", url=url)
        ]]
        await update.message.reply_text(
            f"{emoji} {name[:50]}...\n"
            f"💰 السعر الحالي: {cur:.2f} ج.م\n"
            f"🎯 المستهدف: {target:.2f} ج.م",
            reply_markup=InlineKeyboardMarkup(keyboard)
        )

async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    data = query.data

    if data.startswith("del_"):
        pid = int(data.split("_")[1])
        user_id = query.from_user.id
        conn = get_conn()
        conn.execute("DELETE FROM watchlist WHERE id=? AND user_id=?", (pid, user_id))
        conn.commit()
        conn.close()
        await query.edit_message_text("🗑 تم الحذف!")

# ==================== فحص الأسعار التلقائي ====================
async def check_prices(context: ContextTypes.DEFAULT_TYPE):
    conn = get_conn()
    rows = conn.execute("SELECT id, user_id, url, product_name, current_price, target_price FROM watchlist").fetchall()
    conn.close()

    for row in rows:
        pid, user_id, url, name, old_price, target = row
        name_new, new_price = await get_price(url)
        if new_price is None:
            continue

        conn = get_conn()
        conn.execute("UPDATE watchlist SET current_price=?, last_price=? WHERE id=?", (new_price, old_price, pid))
        conn.commit()
        conn.close()

        # إشعار لو السعر نزل للمستهدف أو أقل
        if new_price <= target and old_price > target:
            try:
                await context.bot.send_message(
                    chat_id=user_id,
                    text=(
                        f"🔥 انخفض السعر!\n\n"
                        f"📦 {name[:60]}\n"
                        f"💰 السعر الجديد: {new_price:.2f} ج.م\n"
                        f"🎯 السعر المستهدف: {target:.2f} ج.م\n\n"
                        f"🛒 اشتري الآن:\n{url}"
                    )
                )
            except Exception as e:
                logger.error(f"Error sending message: {e}")

        # إشعار لو السعر نزل حتى لو مش وصل المستهدف
        elif new_price < old_price:
            diff = old_price - new_price
            try:
                await context.bot.send_message(
                    chat_id=user_id,
                    text=(
                        f"📉 السعر نزل!\n\n"
                        f"📦 {name[:60]}\n"
                        f"💰 السعر القديم: {old_price:.2f} ج.م\n"
                        f"💰 السعر الجديد: {new_price:.2f} ج.م\n"
                        f"✂️ الفرق: {diff:.2f} ج.م\n"
                        f"🎯 المستهدف: {target:.2f} ج.م\n\n"
                        f"🔗 {url}"
                    )
                )
            except Exception as e:
                logger.error(f"Error sending message: {e}")

        await asyncio.sleep(2)  # انتظر بين كل منتج

# ==================== تشغيل البوت ====================
def main():
    init_db()
    app = Application.builder().token(TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("help", help_cmd))
    app.add_handler(CommandHandler("add", add_cmd))
    app.add_handler(CommandHandler("list", list_cmd))
    app.add_handler(CallbackQueryHandler(button_handler))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    # فحص الأسعار كل ساعة
    app.job_queue.run_repeating(check_prices, interval=CHECK_INTERVAL, first=60)

    print("✅ البوت شغال!")
    app.run_polling()

if __name__ == "__main__":
    main()
